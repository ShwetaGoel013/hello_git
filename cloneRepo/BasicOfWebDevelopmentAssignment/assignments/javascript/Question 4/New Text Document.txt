    	let person = {
    		 name : 'Random',
             age : 20,
             address :
          {
          	city : "Delhi",
          	pinCode : 10001
          }
            
    	};
         console.log(person);
         
      
        // shallow copy
    	let cloneOfPerson =Object.assign({},person);
    	cloneOfPerson.name= 'NewName';
    	cloneOfPerson.age=19 ;
    	cloneOfPerson.address.city ="Mumbai";
    	cloneOfPerson.address.pinCode =20002;
    	
    	console.log(cloneOfPerson);




    </script>
